## Basic Usage

<DualCodeSource jsx title="AutoheightWebView" sourceBase={"autoheight"} />

```ts title="autoheight.tsx"
import React, { ComponentProps } from 'react';
import makeWebshell, {
  handleHTMLDimensionsFeature,
  forceResponsiveViewportFeature,
  forceElementSizeFeature,
  useAutoheight
} from '@formidable-webview/webshell';
import WebView from 'react-native-webview';

const Webshell = makeWebshell(
  WebView,
  handleHTMLDimensionsFeature.assemble(),
  forceResponsiveViewportFeature.assemble({ maxScale: 2 }),
  forceElementSizeFeature.assemble({
    target: 'body',
    heightValue: 'auto',
    widthValue: 'auto'
  })
);
```
